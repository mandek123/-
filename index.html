function updateSimPetInfo() { const petImg = document.getElementById('sim-pet-img'); if (!currentSimPet || !petImg) return; // 이미지 경로 설정: 이미지 데이터가 있을 경우 노출 if (currentSimPet.image) { // 경로 예시: ./images/페트파일명.png petImg.src = `./images/${currentSimPet.image}.gif`; petImg.style.display = "block"; // 이미지 로드 실패 시 숨김 처리 (예외처리) petImg.onerror = () => { petImg.style.display = "none"; }; } else { petImg.style.display = "none"; } // 기본 스탯 업데이트 document.getElementById('base-hp').innerText = Number(currentSimPet.hp || 0).toFixed(2); document.getElementById('base-atk').innerText = Number(currentSimPet.atk || 0).toFixed(2); document.getElementById('base-def').innerText = Number(currentSimPet.def || 0).toFixed(2); document.getElementById('base-spd').innerText = Number(currentSimPet.spd || 0).toFixed(2); resetSimUIStats(); }
