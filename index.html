function createPetCard(pet) { const card = document.createElement('div'); card.className = 'res-box'; card.style.display = 'flex'; card.style.flexDirection = 'column'; let gradeClass = 'tag-normal'; if(pet.grade === '최상급') gradeClass = 'tag-top'; if(pet.grade === '상급') gradeClass = 'tag-high'; let coloredElements = pet.elements.split(' ').map(el => { let color = '#d2a8ff'; if (el.includes('지')) color = '#3fb950'; else if (el.includes('수')) color = '#539bf5'; else if (el.includes('화')) color = '#ff7b72'; else if (el.includes('풍')) color = '#e3b341'; return `${el}`; }).join(' '); let passiveDesc = ""; const passiveData = RAID_PASSIVE_DATABASE.find(p => p.name === pet.passive); if (passiveData) { passiveDesc = passiveData.effect; } card.innerHTML = `
${pet.name}
${pet.grade}
${pet.passive && pet.passive !== "없음" && pet.passive !== "미정" ? `${pet.passive}` : ''} ${pet.passive === "미정" ? `패시브 미정` : ''} ${coloredElements}
초기치 (체/공/방/순) ${pet.hp} / ${pet.atk} / ${pet.def} / ${pet.spd}
성장률 (체/공/방/순) ${pet.ghp} / ${pet.gatk} / ${pet.gdef} / ${pet.gspd}
도감능력 (140Lv) ${pet.encyBuff}
`; return card; }
